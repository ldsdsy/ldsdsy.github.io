#include <iostream>
#include <cmath>
using namespace std;

bool HuiWen(int n)
{
	n=n*n;
	int a[10]={0};
	int t=n;
	int i=0;
	while(t!=0)
	{
		a[i]=t%10;
		t/=10;
		i++;
	}
	i--;
	int j=0;
	while(i>=0)
	{
		t+=a[i]*pow(10,j);
		i--;
		j++;
	}
	if(t==n)
	{
		return true;
	}
	return false;
}

int main()
{
	for(int i=0;i<256;i++)
	{
		if(HuiWen(i))
		{
			cout<<i<<" ";
		}
	}
	return 0;
}

#include <iostream>
using namespace std;

int Fibonacci(int j)
{
	if(j==1)
	{
		return 0;
	}
	else if(j==2)
	{
		return 1;
	}
	else
	{
		return Fibonacci(j-1)+Fibonacci(j-2);
	}
}

int main()
{
	int n,t;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		for(int k=1;k<=n-i;k++)
		{
			cout<<"   ";
		}
		for(int j=1;j<=2*i-1;j++)
		{
			t=Fibonacci(j);
			cout<<t<<"  "	;
		}
		cout<<endl;
	}
}

package exam2002b

import "fmt"

//Fibonacci is used to calculate the number
func Fibonacci(j int) int {
	if j == 1 {
		return 0
	} else if j == 2 {
		return 1
	}
	return Fibonacci(j-1) + Fibonacci(j-2)
}

//Triangle is used to print Triangle of Fibonacci
func Triangle(n int) {
	for i := 1; i <= n; i++ {
		for k := 1; k <= n-i; k++ {
			fmt.Print("   ")
		}
		for j := 1; j <= 2*i-1; j++ {
			t := Fibonacci(j)
			fmt.Printf("%d  ", t)
		}
		fmt.Println()
	}
}
