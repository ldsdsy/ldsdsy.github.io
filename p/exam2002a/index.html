package exam2002a

import "fmt"

//Number is used to compute the number of Triangle
func Number(i, j int) int {
	if j == 1 || i == j {
		return 1
	}
	return Number(i-1, j) + Number(i-1, j-1)
}

//Triangle is used to print Yang Hui Triangle
func Triangle(h int) {
	for i := 1; i <= h; i++ {
		for j := 1; j <= h-i; j++ {
			fmt.Print(" ")
		}
		for k := 1; k <= i; k++ {
			t := Number(i, k)
			fmt.Printf("%d ", t)
		}
		fmt.Println()
	}
}
